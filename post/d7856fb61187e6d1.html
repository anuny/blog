        if(unescape(arr[2]) === 'true'){
            var className = document.documentElement.className,hasClass = new RegExp("(\\s|^)color-dark(\\s|$)").test(className);
            !hasClass && (document.documentElement.className += " " + 'color-dark');
            window.isDarked = true
        }
    }
})();
